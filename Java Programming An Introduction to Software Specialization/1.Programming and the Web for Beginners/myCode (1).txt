var image = new SimpleImage("hilton.jpg");
for (var p of image.values()){
     if (p.getX() < (image.getWidth() / 3)){
        p.setRed(255);
    }
    if (p.getX() < (image.getWidth() / 3 * 2) && p.getX() > (image.getWidth() / 3) ){
     p.setGreen(255);
   }
     if(p.getX() > (image.getWidth() / 3 * 2)){
        p.setBlue(255);
    }
    
}
print(image);
